bool isPointInPolygon(P p, vp &a){
    int w=0;
    rep(i, n){
        int k=sgn((a[i+1]-a[i])*(p-a[i]));
        int d1=sgn(a[i].y-p.y);
        int d2=sgn(a[i+1].y-p.y);
        if (k>0 && d1<=0 && d2>0) w++;
        if (k<0 && d2<=0 && d1>0) w--;
    }
    if (w!=0) return 1;
    return 0;
}
